SELECT batchid,
       jsonfilename,
       json_processing_status,
       iics_processing_status,
       startdatetime,
       enddatetime,
       comments
FROM   clg_reporting.los_pcg.du_json_sourcefile List;

UPDATE clg_reporting.los_pcg.du_json_sourcefilelist
SET    iics_processing_status = 'COMPLETED',
       enddatetime = CURRENT_TIMESTAMP,
       comments = 'IICS Load Successful'
WHERE  json_processing_status = 'COMPLETED'
       AND iics_processing_status = 'OPEN';

DELETE FROM clg_reporting.los_pcg.du_documents
WHERE  batchid IN (SELECT batchid
                   FROM   clg_reporting.los_pcg.du_json_sourcefilelist
                   WHERE  json_processing_status = 'COMPLETED'
                          AND iics_processing_status = 'OPEN');

DELETE FROM clg_reporting.los_pcg.du_borrower
WHERE  batchid IN (SELECT batchid
                   FROM   clg_reporting.los_pcg.du_json_sourcefilelist
                   WHERE  json_processing_status = 'COMPLETED'
                          AND iics_processing_status = 'OPEN');

DELETE FROM clg_reporting.los_pcg.du_message_data
WHERE  batchid IN (SELECT batchid
                   FROM   clg_reporting.los_pcg.du_json_sourcefilelist
                   WHERE  json_processing_status = 'COMPLETED'
                          AND iics_processing_status = 'OPEN');

DELETE FROM clg_reporting.los_pcg.du_messages
WHERE  batchid IN (SELECT batchid
                   FROM   clg_reporting.los_pcg.du_json_sourcefilelist
                   WHERE  json_processing_status = 'COMPLETED'
                          AND iics_processing_status = 'OPEN');

DELETE FROM los_pcg.du_borrower_stg;

DELETE FROM los_pcg.du_messages_stg;

DELETE FROM los_pcg.du_message_data_stg;

DELETE FROM los_pcg.du_documents_stg;

DELETE FROM los_pcg.du_loandata_stg;

DELETE FROM los_pcg.du_json_sourcefilelist;

FROM   los_pcg.du_borrower_stg;

FROM   los_pcg.du_messages_stg;

FROM   los_pcg.du_message_data_stg;

FROM   los_pcg.du_documents_stg;

FROM   los_pcg.du_loandata_stg;

FROM   los_pcg.du_json_sourcefilelist;
